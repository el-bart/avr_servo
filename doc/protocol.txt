host->device:
  <servo_name><mode><pos1><pos2><end>
  where:
    servo_name is 'a'..'r'
    mode is 'l' (lowest), 'h' (highest), 'd' (default), 's' (set)
    pos[12] are 1 byte encoded as 2 pritable hex digits
    end is CR or LF (note: any sequence of CR/LF will threated as the end of line)

device->host:
  <servo_no>OK<end>
  where:
    servo_no is 'a'..'r'
    end is CR
